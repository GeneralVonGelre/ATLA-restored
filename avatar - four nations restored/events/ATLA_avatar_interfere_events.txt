letter_event = { #Avatar makes peace
	id = NAE.201
	desc = NAE.201.D
	portrait = persistent_event_target:the_avatar
	border = GFX_event_letter_frame_avatar
	
	only_independent = yes
	war = yes
	only_capable = yes
	has_global_flag = avatar_available
	
	is_triggered_only = yes #Fired bi-yearly in the 'Avatar intereference' block
	
	trigger = {
		has_tyrannical_personality_trigger = no
		NAND = {
			ai = yes
			any_current_enemy = { ai = yes }
		}
		is_hostile_to_avatar_trigger = no
		in_revolt = no
		is_primary_war_attacker = yes #Since revolters are always the aggressors, this lets us filter only to people warring on other rulers
		NOT = { any_realm_lord = { trait = avatar } }
	}
	
	weight_multiplier = {
		days = 1
		modifier = {
			factor = 1.2
			persistent_event_target:the_avatar = { has_education_diplomacy_trigger = yes }
		}
		modifier = {
			factor = 1.1
			persistent_event_target:the_avatar = { has_benevolent_trait_trigger = yes }
		}
		modifier = {
			factor = 1.2
			persistent_event_target:the_avatar = { is_benevolent_trigger = yes }
		}
		modifier = {
			factor = 1.2
			persistent_event_target:the_avatar = { has_pleasant_trait_trigger = yes }
		}
		modifier = {
			factor = 1.4
			persistent_event_target:the_avatar = { pacifist = yes }
		}
		modifier = {
			factor = 0
			persistent_event_target:the_avatar = {
				OR = {
					ai = no
					trait = bad_avatar
					has_character_flag = withdrawn_avatar
				}
			}
		}
	}
	
	option = { #Agree to peace
		name = NAE.201.OPTA
		any_war = { end_war = whitepeace }
		remove_trait = ambitious
		add_character_modifier = {
			name = avatars_peace
			years = 5
		}
		add_trait = avatars_peace_trait
		character_event = { id = NAE.2061 years = 5 } #Remove trait
		
		any_player = { character_event = { id = NAE.501 } }
		
		ai_chance = {
			factor = 70
			modifier = { #Scales with Avatar's diplomacy score
				persistent_event_target:the_avatar = { diplomacy = 12 }
				factor = 1.1
			}
			modifier = {
				persistent_event_target:the_avatar = { diplomacy = 16 }
				factor = 1.2
			}
			modifier = {
				persistent_event_target:the_avatar = { diplomacy = 21 }
				factor = 1.3
			}
			modifier = {
				persistent_event_target:the_avatar = { diplomacy = 26 }
				factor = 1.4
			}
			modifier = {
				persistent_event_target:the_avatar = { diplomacy = 31 }
				factor = 1.5
			}
		}
	}
	
	option = { #Refuse to end wars
		name = NAE.201.OPTB
		add_rival = persistent_event_target:the_avatar
		
		ai_chance = {
			factor = 30
			modifier = {
				has_tyrannical_personality_trigger = yes
				factor = 5
			}
			modifier = {
				persistent_event_target:the_avatar = { diplomacy < 8 }
				factor = 2
			}
			modifier = {
				factor = 2
				trait = stubborn
			}
			modifier = {
				tier = EMPEROR
				factor = 3
			}
		}
	}
	
	after = {
		clr_global_flag = avatar_available
		persistent_event_target:the_avatar = { character_event = { id = NAE.200 } } #Avatar wander prompt
	}
}
